### 多线程

[1114. 按序打印](https://leetcode.cn/problems/print-in-order/)

使用条件变量 `std::condition_variable`即可

```c++
#include <condition_variable>
#include <mutex>
class Foo {
    std::mutex mtx;
    std::condition_variable cv;
    int k = 1; // 状态变量：1 代表一，2 代表二，3 代表三
public:
    Foo() {}

    void first(function<void()> printFirst) {
        std::unique_lock<std::mutex> lock(mtx);
        // first 不需要等待，直接打印
        // printFirst() outputs "first". Do not change or remove this line.
        printFirst();
        k = 2;           // 更新状态
        cv.notify_all(); // 唤醒正在等待的线程（second 或 third）
    }

    void second(function<void()> printSecond) {
        std::unique_lock<std::mutex> lock(mtx);
        // 如果不是 2，就一直等
        cv.wait(lock, [this]() { return k == 2; });

        // printSecond() outputs "second". Do not change or remove this line.
        printSecond();
        k = 3;           // 更新状态
        cv.notify_all(); // 唤醒正在等待的 third
    }

    void third(function<void()> printThird) {
        std::unique_lock<std::mutex> lock(mtx);
        // 如果不是 3，就一直等
        cv.wait(lock, [this]() { return k == 3; });

        // printThird() outputs "third". Do not change or remove this line.
        printThird();
    }
};
```

[1115. 交替打印 FooBar](https://leetcode.cn/problems/print-foobar-alternately/)

```c++
#include <mutex>
#include <condition_variable>
class FooBar {
private:
    int n;
    std::mutex mtx;
    std::condition_variable cv;
    bool foo_done = false; // 标志位：false 代表该打印 foo，true 代表该打印 bar
public:
    FooBar(int n) {
        this->n = n;
    }

    void foo(function<void()> printFoo) {
        
        for (int i = 0; i < n; i++) {
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock, [this]() { return !foo_done; });
        	// printFoo() outputs "foo". Do not change or remove this line.
        	printFoo();
            foo_done = true;  // 标记 foo 已完成
            cv.notify_one();  // 唤醒 bar 线程
        }
    }

    void bar(function<void()> printBar) {
        
        for (int i = 0; i < n; i++) {
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock, [this]() { return foo_done; });
        	// printBar() outputs "bar". Do not change or remove this line.
        	printBar();
            foo_done = false;  // 标记 bar 已完成
            cv.notify_one();  // 唤醒 foo 线程
        }
    }
};
