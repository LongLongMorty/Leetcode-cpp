[1456. 定长子串中元音的最大数目](https://leetcode.cn/problems/maximum-number-of-vowels-in-a-substring-of-given-length/)

```c++
class Solution {
public:
    bool is_vo(char c) {
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
            return true;
        }
        return false;
    }
    int maxVowels(string s, int k) {
        int r = 0;
        int temp = 0;
        for (int i = 0; i < k; i++) {
            if (is_vo(s[i])) {
                temp++;
            }
        }
        r = temp;
        for (int i = k; i < s.size(); i++) {
            if (is_vo(s[i - k])) {
                temp--;
            }
            if (is_vo(s[i])) {
                temp++;
            }
            if (temp >= r) {
                r = temp;
            }
        }
        return r;
    }
};
```

[643. 子数组最大平均数 I](https://leetcode.cn/problems/maximum-average-subarray-i/)

```c++
class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        double r = 0;
        double sum = 0;
        for (int i = 0; i < k; i++) {
            sum += nums[i];
        }
        r = sum;
        for (int i = k; i < nums.size(); i++) {
            sum -= nums[i - k];
            sum += nums[i];
            if (sum > r) {
                r = sum;
            }
        }
        return r/k;
    }
};
```

```c++
class Solution {

        for (int i = 0; i < k; i++) {
            sum += arr[i];

        if (sum / k >= threshold) {
            count++;
        }

            }
        }
        return count;
    }
```

[2841. 几乎唯一子数组的最大和](https://leetcode.cn/problems/maximum-sum-of-almost-unique-subarray/)

哈希思路解决，用个map判断当前窗口有多少种元素及其个数

```c++
class Solution {
public:
    long long maxSum(vector<int>& nums, int m, int k) {
        unordered_map<int,int> us;
        long long sum = 0;
        long long r = 0;
        for(int i = 0;i < k;i++){
             us[nums[i]]++;
             sum += nums[i];
        }
        if(us.size() >= m){
            r = sum;
        }
        for(int i = k;i < nums.size();i++){
            sum -= nums[i-k];
            sum += nums[i];
            us[nums[i-k]]--;
            if(us[nums[i-k]] == 0){
                us.erase(nums[i-k]);
            }
             us[nums[i]]++;
             if(us.size() >= m && sum >= r){
                r = sum;
             }
        }
        return r;
    }
};
```

[2379. 得到 K 个黑块的最少涂色次数](https://leetcode.cn/problems/minimum-recolors-to-get-k-consecutive-black-blocks/)

可以直接用count 计数黑块数量

```c++
class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        int n = blocks.size();
        int countB = 0;

        // 第一个窗口
        for (int i = 0; i < k; i++) {
            if (blocks[i] == 'B')
                countB++;
        }

        int maxB = countB;
        if (maxB == k) {
            return 0;
        }

        // 滑动窗口
        for (int i = k; i < n; i++) {
            // 去掉左边
            if (blocks[i - k] == 'B')
                countB--;
            // 加入右边
            if (blocks[i] == 'B')
                countB++;

            maxB = max(maxB, countB);
            if (maxB == k) {
                return 0;
            }
        }

        return k - maxB;
    }
};
```

[1423. 可获得的最大点数](https://leetcode.cn/problems/maximum-points-you-can-obtain-from-cards/)

我按题目引导，想到的就是前面拿`x`个，后面拿`k-x`个，暴力解法超时，就考虑前后缀求解

```c++
class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int sum = 0;
        int r = 0;
        vector<int> pre(cardPoints.size(), 0);
        vector<int> post(cardPoints.size(), 0);
        pre[0] = cardPoints[0];
        post[cardPoints.size() - 1] = cardPoints[cardPoints.size() - 1];
        for (int i = 1; i < cardPoints.size(); i++) {
            pre[i] = cardPoints[i] + pre[i - 1];
        }
        for (int i = cardPoints.size() - 2; i >= 0; i--) {
            post[i] = cardPoints[i] + post[i + 1];
        }
        for (int i = 0; i <= k; i++) {
            if (i != 0) {
                sum += pre[i - 1];
            }
            if (i != k) {
                sum += post[cardPoints.size() - (k - i)];
            }
            if (sum > r) {
                r = sum;
            }
            sum = 0;
        }
        return r;
    }
};
```

后面看了题解发现，其实就是普通滑动窗口换了问法

```c++
class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        int total = accumulate(cardPoints.begin(), cardPoints.end(), 0);
        if (k == n) return total;

        int window = n - k;
        int cur = 0;
        for (int i = 0; i < window; i++) cur += cardPoints[i];
        int minSum = cur;

        for (int i = window; i < n; i++) {
            cur += cardPoints[i] - cardPoints[i - window];
            minSum = min(minSum, cur);
        }
        return total - minSum;
    }
};
```

[1052. 爱生气的书店老板](https://leetcode.cn/problems/grumpy-bookstore-owner/)

先算基础满意人数，然后找到最长连续时间内最大不满意人数，满意人数加上这些不满意人数就是答案

```c++
class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int minutes) {
        int sati = 0;
        int r = 0;
        int unsati =0;
        for(int i = 0;i < minutes;i++){
            if(grumpy[i] == 1){
                unsati += customers[i];
            }else{
                sati += customers[i];
            }
        }
        r = unsati;
        for(int i = minutes;i < customers.size();i++){
            if(grumpy[i] == 1){
                unsati += customers[i];
            }else{
                 sati += customers[i];
            }
            if(grumpy[i-minutes] == 1){
                unsati -= customers[i-minutes];
            }
            if(unsati > r){
                r = unsati;
            }
        }
        return r + sati;
    }
};
```

[567. 字符串的排列](https://leetcode.cn/problems/permutation-in-string/)

一开始我只想到了map存储然后对比，但是太费内存和时间了

```c++
class Solution {
public:
    bool is_check(unordered_map<char, int> um, int l1, const string& s2,
                  int i) {
        for (int j = i; j < i + l1; j++) {
            if (um.count(s2[j])) {
                if (um[s2[j]] == 1) {
                    um.erase(s2[j]);
                } else {
                    um[s2[j]]--;
                }
            } else {
                return false;
            }
        }
        if (um.empty()) {
            return true;
        }
        return false;
    }
    bool checkInclusion(string s1, string s2) {
        int l1 = s1.size();
        if (s1.size() > s2.size()) {
            return false;
        }
        unordered_map<char, int> um;
        for (int i = 0; i < l1; i++) {
            um[s1[i]]++;
        }
        for (int i = 0; i <= s2.size() - l1; i++) {
            if (is_check(um, l1, s2, i)) {
                return true;
            }
        }
        return false;
    }
};
```

```c++
class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int l1 = s1.size();
        if (s1.size() > s2.size()) {
            return false;
        }
        vector<int> a(26, 0), b(26, 0);
        for (int i = 0; i < l1; i++) {
            a[s1[i] - 'a']++;
            b[s2[i] - 'a']++;
        }
        if (a == b) {
            return true;
        }
        for (int i = l1; i < s2.size(); i++) {
            b[s2[i] - 'a']++;
            b[s2[i - l1] - 'a']--;
            if (a == b) {
                return true;
            }
        }
        return false;
    }
};
```

[941. 有效的山脉数组](https://leetcode.cn/problems/valid-mountain-array/)

不会做，直接看答案才做出来了

```c++
class Solution {
public:
    bool validMountainArray(vector<int>& arr) {
        int N = arr.size();
        int i = 0;

        // 递增扫描
        while (i + 1 < N && arr[i] < arr[i + 1]) {
            i++;
        }

        // 最高点不能是数组的第一个位置或最后一个位置
        if (i == 0 || i == N - 1) {
            return false;
        }

        // 递减扫描
        while (i + 1 < N && arr[i] > arr[i + 1]) {
            i++;
        }

        return i == N - 1;
    }
};
```
