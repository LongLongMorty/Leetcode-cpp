[1456. 定长子串中元音的最大数目](https://leetcode.cn/problems/maximum-number-of-vowels-in-a-substring-of-given-length/)

```c++
class Solution {
public:
    bool is_vo(char c) {
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
            return true;
        }
        return false;
    }
    int maxVowels(string s, int k) {
        int r = 0;
        int temp = 0;
        for (int i = 0; i < k; i++) {
            if (is_vo(s[i])) {
                temp++;
            }
        }
        r = temp;
        for (int i = k; i < s.size(); i++) {
            if (is_vo(s[i - k])) {
                temp--;
            }
            if (is_vo(s[i])) {
                temp++;
            }
            if (temp >= r) {
                r = temp;
            }
        }
        return r;
    }
};
```

[643. 子数组最大平均数 I](https://leetcode.cn/problems/maximum-average-subarray-i/)

```c++
class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        double r = 0;
        double sum = 0;
        for (int i = 0; i < k; i++) {
            sum += nums[i];
        }
        r = sum;
        for (int i = k; i < nums.size(); i++) {
            sum -= nums[i - k];
            sum += nums[i];
            if (sum > r) {
                r = sum;
            }
        }
        return r/k;
    }
};
```

[1343. 大小为 K 且平均值大于等于阈值的子数组数目](https://leetcode.cn/problems/number-of-sub-arrays-of-size-k-and-average-greater-than-or-equal-to-threshold/)

```c++
class Solution {
public:
    int numOfSubarrays(vector<int>& arr, int k, int threshold) {
        double sum = 0;
        for (int i = 0; i < k; i++) {
            sum += arr[i];
        }
        int count = 0;
        if (sum / k >= threshold) {
            count++;
        }
        for (int i = k; i < arr.size(); i++) {
            sum -= arr[i - k];
            sum += arr[i];
            if (sum / k >= threshold) {
                count++;
            }
        }
        return count;
    }
};
```

[2841. 几乎唯一子数组的最大和](https://leetcode.cn/problems/maximum-sum-of-almost-unique-subarray/)

哈希思路解决，用个map判断当前窗口有多少种元素及其个数

```c++
class Solution {
public:
    long long maxSum(vector<int>& nums, int m, int k) {
        unordered_map<int,int> us;
        long long sum = 0;
        long long r = 0;
        for(int i = 0;i < k;i++){
             us[nums[i]]++;
             sum += nums[i];
        }
        if(us.size() >= m){
            r = sum;
        }
        for(int i = k;i < nums.size();i++){
            sum -= nums[i-k];
            sum += nums[i];
            us[nums[i-k]]--;
            if(us[nums[i-k]] == 0){
                us.erase(nums[i-k]);
            }
             us[nums[i]]++;
             if(us.size() >= m && sum >= r){
                r = sum;
             }
        }
        return r;
    }
};
```

[2461. 长度为 K 子数组中的最大和](https://leetcode.cn/problems/maximum-sum-of-distinct-subarrays-with-length-k/)

思路和上一道几乎一样，感觉更简单

```c++
class Solution {
public:
    long long maximumSubarraySum(vector<int>& nums, int k) {
        unordered_map<int,int> us;
        long long sum = 0;
        long long r = 0;
        for(int i = 0;i < k;i++){
             us[nums[i]]++;
             sum += nums[i];
        }
        if(us.size() == k){
            r = sum;
        }
        for(int i = k;i < nums.size();i++){
            sum -= nums[i-k];
            sum += nums[i];
            us[nums[i-k]]--;
            if(us[nums[i-k]] == 0){
                us.erase(nums[i-k]);
            }
             us[nums[i]]++;
             if(us.size() == k && sum >= r){
                r = sum;
             }
        }
        return r;
    }
};
```
